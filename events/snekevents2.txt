		months = 60
		#days = 10
	}
	pre_triggers = {

}
	trigger = {
		owner = {
			exists = this
			has_country_flag = grand_campaign_player_one
			is_at_war_with = event_target:coc_target
		}
		num_pops > 3
		any_owned_pop = {
			has_ethic = ethic_spiritualist
		}
	}
	
	immediate = {
		random_owned_pop = {
			species = {
				save_event_target_as = rebel_species
			}
		}
	}
	
	
	option = {
		name = snek2.1.a
		if = {
			limit = {
				num_pops > 30
			}
			while = {
				count = 15
				create_army = {
						owner = event_target:coc_target
						species = event_target:rebel_species
						type = "assault_army"
				}
			}
		}
		else_if = {
			limit = {
				num_pops > 20
			}
			while = {
				count = 9
				create_army = {
						owner = event_target:coc_target
						species = event_target:rebel_species
						type = "assault_army"
				}
			}
		}
		else_if = {
			limit = {
				num_pops > 10
			}
			while = {
				count = 5
				create_army = {
						owner = event_target:coc_target
						species = event_target:rebel_species
						type = "assault_army"
				}
			}
		}
		else = {
			while = {
				count = 4
				create_army = {
						owner = event_target:coc_target
						species = event_target:rebel_species
						type = "assault_army"
				}
			}
